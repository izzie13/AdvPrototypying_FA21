Project 3: Data System

Summary

In this project, I used two vibration sensors and one photocell to determine the noise in the room related to the time by detecting the vibration of the window. And I use a grid of squares to visualize the tensity of vibration and use the color to show the day and night.

let intID;

let value_vib2 = 0;
let value_light = 0;function setup() {
 createCanvas(800, 800);
 getData();
 background(200, 200, 300, 200);
 // marginX = 600 - int(600 / horizon) * horizon;
 // marginY = 400 - int(400 / horizon) * horizon;
 // rectMode(CENTER);
 intID = setInterval(getData, 3000);
 colorMode(HSB,200);
}function draw() {
 horizon = 230 - value_vib2;
 marginX = 600 - int(600 / horizon) * horizon;
 marginY = 400 - int(400 / horizon) * horizon;
 rectMode(CENTER);
 background(value_light/2, value_light*1.1, 190, value_light*2);
 stroke(value_light/2, value_light, 120, 90);
 fill(value_light*2, value_light*1.5, value_light/0.05, 160);
 strokeWeight(3);  for (
   let i = marginX / vertical + horizon / vertical;
   i < width - marginX / vertical;
   i += horizon
 ) {
   for (
     let j = marginY / 2 + horizon / 2;
     j < height - marginY / 2;
     j += horizon
   ) {
     // let angle = map(30, 0, width, 0, TWO_PI, true);
     push();
     translate(i, j);
     rotate(value_vib1);
     square(0, 0, value_vib2);
     pop();
   }
 }
}function getData() {
 //callback is important, regain the control
 feed_vib1 = loadJSON(
   "https://io.adafruit.com/api/v2/izzie_c/feeds/vib1-data-feed/data?limit=1",
   processVib1, handleError
 );
 feed_vib2 = loadJSON(
   "https://io.adafruit.com/api/v2/izzie_c/feeds/vib2-data-feed/data?limit=1",
   processVib2, handleError
 );
 feed_light = loadJSON(
   "https://io.adafruit.com/api/v2/izzie_c/feeds/photo-data-feed/data?limit=1",
   processLight, handleError
 );
}function processVib2(feedData) {
 console.log(feedData);
 //value_vib1 = map(feedData[0].value,0,4095,0,200);
 value_vib2 = map(feedData[0].value,90,150,0,200);
}
function processVib1(feedData) {
 console.log(feedData);
 //value_vib2 = feedData[0].value;
 value_vib1 = map(feedData[0].value,50,200,0,TWO_PI/3);
}
function processLight(feedData) {
 console.log(feedData);
 //value_light = feedData[0].value;
 value_light = map(feedData[0].value,600,630,10,80);
}
function handleError(error){
 console.log(error);
